# printf
> stdio.h에서 제공해주는 c언어의 대표적인 문장을 출력하는 기능입니다.
> 일반적으로는 printf("작성하고 싶은 문장")을 작성해 사용합니다.

예시)
~~~c
int main()
{
   printf("프로그래밍")
   return 0;
}

~~~
